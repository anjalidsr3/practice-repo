       var count=1;

       function changeStyle(counter){
      if (count<=9)
      {
        if (count%2!=0)
      {
        document.getElementById(counter.id).innerHTML="X";
      }else
        {
           document.getElementById(counter.id).innerHTML="O";
        }
         count++;
       }else
         {
           
           reset();
         }
      }
      function reset()
      {
        for (var i=1; i<=9; i++)
          {
            document.getElementById('div'+i).innerHTML="";
          }
      }
